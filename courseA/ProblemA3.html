課題A3では，JHotDrawの図形選択機能と図形移動機能の間の依存関係について見ていきます． JHotDrawでは，選択モード時にマウスクリックして選択状態にした図形を，ドラッグして任意の位置に移動することができます．これは, JHotDrawにおいて図形移動機能が図形選択機能に依存していることを意味しています． ここでは，図形選択機能で実行されるソースコードと図形移動機能で実行されるソースコードの中から, これらの間の依存関係に関係する部分を，ある自動抽出技術を用いて抜き出したものについて見ていきましょう． 図形選択機能を実行すると．SelectionToolクラスのmousePressed()メソッド(以下，SelectionTool#mousePressed()のように書く)が最初に呼び出されます． 一方，図形選択機能の実行後に図形移動機能を実行すると，マウスボタン押下時にSelectionTool#mousePressed()メソッドが呼び出された後，マウスドラッグ中にSelectionTool#mouseDragged()メソッドが繰り返し呼び出されます． まず，SelectionToolクラスとその親クラスの図形選択機能に関係する部分を以下に示します．

public abstract class AbstractTool extends AbstractBean implements Tool {
    protected DrawingEditor editor;
    protected Point anchor;

    protected DrawingView getView() {
        return editor.getActiveView();
    } //呼び出し元に戻る
}

public class SelectionTool extends AbstractTool {
    private Tool tracker;

    public void mouseDragged(MouseEvent evt) {
        tracker.mouseDragged(evt);
    }

    public void mousePressed(MouseEvent evt) {
        Tool newTracker;
        Figure figure;
        newTracker = getDragTracker(figure);
        setTracker(newTracker);
        tracker.mousePressed(evt);
    }

    protected void setTracker(Tool newTracker) {
        tracker = newTracker;
    } //呼び出し元に戻る

    protected DragTracker getDragTracker(Figure f) {
        dragTracker = new DefaultDragTracker();
        return dragTracker;
    } //呼び出し元に戻る
}

public class DefaultDragTracker extends AbstractTool implements DragTracker {
    private HashSet<Figure> transformedFigures;

    public void mousePressed(MouseEvent evt) {
        DrawingView view = getView();
        transformedFigures = new HashSet<Figure>();
        for (Figure f : view.getSelectedFigures()) {
            transformedFigures.add(f);
        } //呼び出し元に戻る
    }

    public void mouseDragged(MouseEvent evt) {
        AffineTransform tx;
        for (Figure f : transformedFigures) {
            f.transform(tx);
        }
    } //呼び出し元に戻る
}
